Implementación en R

# Parámetros del GCL
a <- 1103515245
c <- 12345
m <- 2^31
semilla <- 1234  # Últimos 4 dígitos de tu cédula

# Implementación
gcl_mixto <- function(n, a, c, m, seed) {
  x <- numeric(n)
  x[1] <- seed
  for(i in 2:n) {
    x[i] <- (a * x[i-1] + c) %% m
  }
  return(x/m)
}

# Generar 10,000 números
numeros <- gcl_mixto(10000, a, c, m, semilla)

# Histograma
hist(numeros, breaks=10, main="Distribución de Números Pseudoaleatorios", 
     xlab="Valor", ylab="Frecuencia", col="lightblue")
